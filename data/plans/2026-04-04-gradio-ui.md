## Key Design Decisions

### Person Selection in Gradio

The CLI uses `xdg-open` + `input()` for person selection. For Gradio, we use a **click-on-preview** pattern:

1. User uploads video → clicks **"Detect Persons"**
2. Backend calls `RTMPoseExtractor.preview_persons()` → returns list of persons with bboxes
3. Backend renders first frame with numbered bounding boxes (colored rectangles + labels)
4. Annotated frame shown in `gr.Image` component
5. User clicks on a person → `gr.Image.select()` returns `(x, y)` pixel coordinates
6. Backend matches click against person bounding boxes → selects closest person
7. Selected person highlighted with green border → ready for processing

### State Management

Use `gr.State()` to persist between steps:
- `preview_state`: list of detected persons (from `preview_persons()`)
- `person_click_state`: selected `PersonClick` object
- `extractor_state`: initialized `RTMPoseExtractor` instance (reuse GPU model)

### Video Processing

- Use `gr.Progress(track_tqdm=True)` to show frame-by-frame progress
- Process on GPU (same as CLI) — `concurrency_limit=1` to avoid GPU contention
- Output: `gr.Video` player + `gr.File` download buttons for NPY/CSV

```python
"""Pure helper functions for the Gradio UI.

All functions are stateless and testable without a running Gradio server.
"""

from __future__ import annotations

import json
from pathlib import Path

import cv2
import numpy as np
from numpy.typing import NDArray

from src.types import PersonClick
from src.utils.video import get_video_meta


def match_click_to_person(
    persons: list[dict],
    x: float,
    y: float,
) -> dict | None:
    """Match a normalized click coordinate to the closest person bbox.

    Args:
        persons: List from ``preview_persons()``, each with ``bbox`` and ``mid_hip``.
        x: Click x in normalized [0, 1].
        y: Click y in normalized [0, 1].

    Returns:
        Best matching person dict, or None if click is outside all bboxes.
    """
    if not persons:
        return None

    best: dict | None = None
    best_dist = float("inf")

    for p in persons:
        x1, y1, x2, y2 = p["bbox"]
        # Check if click is inside bbox
        if x1 <= x <= x2 and y1 <= y <= y2:
            mx, my = p["mid_hip"]
            dist = (x - mx) ** 2 + (y - my) ** 2
            if dist < best_dist:
                best_dist = dist
                best = p

    return best


def render_person_preview(
    frame: NDArray[np.uint8],
    persons: list[dict],
    selected_idx: int | None = None,
) -> NDArray[np.uint8]:
    """Draw numbered bounding boxes for each detected person.

    Args:
        frame: BGR frame (H, W, 3).
        persons: List from ``preview_persons()``.
        selected_idx: If set, draw this person's box in green instead of blue.

    Returns:
        Annotated frame (same shape).
    """
    if not persons:
        return frame

    annotated = frame.copy()
    h, w = frame.shape[:2]

    colors = [
        (255, 165, 0),   # Blue (OpenCV BGR)
        (0, 200, 200),   # Yellow
        (200, 100, 0),   # Cyan
        (200, 0, 200),   # Magenta
        (0, 180, 255),   # Orange
    ]

    for i, p in enumerate(persons):
        x1, y1, x2, y2 = p["bbox"]
        px1, py1 = int(x1 * w), int(y1 * h)
        px2, py2 = int(x2 * w), int(y2 * h)

        if selected_idx is not None and i == selected_idx:
            color = (0, 255, 0)  # Green for selected
            thickness = 3
        else:
            color = colors[i % len(colors)]
            thickness = 2

        cv2.rectangle(annotated, (px1, py1), (px2, py2), color, thickness)

        # Number label
        label = f"#{i + 1} (hits: {p['hits']})"
        cv2.rectangle(annotated, (px1, py1 - 28), (px1 + len(label) * 10 + 10, py1), color, -1)
        cv2.putText(
            annotated, label, (px1 + 5, py1 - 8),
            cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 255), 1, cv2.LINE_AA,
        )

    return annotated


def persons_to_choices(persons: list[dict]) -> list[str]:
    """Convert person list to Gradio Radio choices.

    Returns:
        List like ["Person #1 (30 hits, track 0)", ...]
    """
    return [
        f"Person #{i + 1} ({p['hits']} hits, track {p['track_id']})"
        for i, p in enumerate(persons)
    ]


def choice_to_person_click(
    choice: str,
    persons: list[dict],
    width: int,
    height: int,
) -> PersonClick:
    """Convert a Gradio Radio selection to a PersonClick.

    Args:
        choice: Radio selection string like "Person #1 (30 hits, track 0)".
        persons: Original person list from preview_persons().
        width: Frame width for coordinate conversion.
        height: Frame height for coordinate conversion.

    Returns:
        PersonClick with pixel coordinates.
    """
    idx = int(choice.split("#")[1].split(" ")[0]) - 1
    mid_hip = persons[idx]["mid_hip"]
    return PersonClick(
        x=int(mid_hip[0] * width),
        y=int(mid_hip[1] * height),
    )


def process_video_pipeline(
    video_path: str,
    person_click: PersonClick | None,
    layer: int,
    tracking: str,
    use_3d: bool,
    render_scale: float,
    blade_3d: bool,
    export: bool,
    output_path: str,
    progress_cb=None,
) -> dict:
    """Run the full visualization pipeline (mirrors visualize_with_skeleton.py).

    Args:
        video_path: Input video file path.
        person_click: Selected person or None (auto-detect single person).
        layer: HUD layer 0-3.
        tracking: Tracking mode ("auto", "sports2d", "deepsort").
        use_3d: Enable 3D pose correction.
        render_scale: Downscale factor for rendering.
        blade_3d: Enable 3D blade detection.
        export: Export poses NPY + biomechanics CSV.
        output_path: Where to save output video.
        progress_cb: Optional callable(progress_frac, desc) for Gradio progress.

    Returns:
        Dict with keys: "video_path", "poses_path", "csv_path", "stats".
    """
    import csv as _csv
    import time as _time
    from collections import Counter

    from src.analysis.angles import compute_joint_angles
    from src.pose_estimation import H36Key
    from src.pose_estimation.rtmlib_extractor import RTMPoseExtractor
    from src.utils.smoothing import PoseSmoother, get_skating_optimized_config
    from src.visualization import (
        JointAngleLayer,
        LayerContext,
        TrailLayer,
        VelocityLayer,
        VerticalAxisLayer,
        draw_skeleton,
        put_text,
        render_layers,
    )

    video_path = Path(video_path)
    output_path = Path(output_path)

    meta = get_video_meta(str(video_path))
    cap = cv2.VideoCapture(str(video_path))

    if progress_cb:
        progress_cb(0.0, "Extracting poses...")

    # Extract poses
    extractor = RTMPoseExtractor(
        output_format="normalized",
        conf_threshold=0.3,
        det_frequency=1,
        device="cuda",
        tracking_mode=tracking,
    )
    extraction = extractor.extract_video_tracked(str(video_path), person_click=person_click)

    raw_poses = extraction.poses
    raw_foot_kps = extraction.foot_keypoints
    n_valid = int(extraction.valid_mask().sum())

    poses_norm = raw_poses[:, :, :2].copy()
    confs = raw_poses[:, :, 2].copy()
    poses = raw_poses.copy()
    poses[:, :, 0] *= meta.width
    poses[:, :, 1] *= meta.height
    pose_frame_indices = np.arange(len(poses))

    # Smoothing
    if len(poses_norm) > 2:
        smooth_config = get_skating_optimized_config(meta.fps)
        smoother = PoseSmoother(smooth_config, freq=meta.fps)
        poses_norm = smoother.smooth(poses_norm)

    poses_viz = poses_norm

    # 3D (if requested)
    poses_3d = None
    if use_3d:
        from src.pose_3d.biomechanics_estimator import Biomechanics3DEstimator
        estimator = Biomechanics3DEstimator()
        poses_3d = estimator.estimate_3d(poses_viz)

    if progress_cb:
        progress_cb(0.3, "Poses extracted. Rendering...")

    # Render scale
    out_w = int(meta.width * render_scale)
    out_h = int(meta.height * render_scale)
    fourcc = cv2.VideoWriter_fourcc(*"mp4v")
    writer = cv2.VideoWriter(str(output_path), fourcc, meta.fps, (out_w, out_h))

    # Build layers
    layers: list = []
    if layer >= 1:
        layers.append(VelocityLayer(scale=3.0, max_length=30, color_mode="solid"))
        layers.append(
            TrailLayer(length=20, joint=H36Key.LFOOT, width=1, color=(200, 80, 80))
        )
        from src.visualization.layers.joint_angle_layer import DEFAULT_JOINT_SPECS, JointAngleSpec
        big_arc_specs = [
            JointAngleSpec(
                s.name, s.point_a, s.vertex, s.point_c,
                s.color, arc_radius=22, good_range=s.good_range, warn_range=s.warn_range,
            )
            for s in DEFAULT_JOINT_SPECS
        ]
        layers.append(JointAngleLayer(joints=big_arc_specs, show_degree_labels=True))
    if layer >= 2:
        layers.append(VerticalAxisLayer())

    # Export buffers
    export_frames: list[int] = []
    export_timestamps: list[float] = []
    export_floor_angles: list[float] = []
    export_joint_angles: list[dict[str, float]] = []
    export_poses_list: list[np.ndarray] = []

    # Frame-by-frame rendering
    frame_idx = 0
    pose_idx = 0
    total = meta.num_frames

    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break

        if render_scale != 1.0:
            frame = cv2.resize(frame, (out_w, out_h), interpolation=cv2.INTER_LINEAR)
            draw_h, draw_w = out_h, out_w
        else:
            draw_h, draw_w = meta.height, meta.width

        current_pose_idx = None
        while pose_idx < len(pose_frame_indices):
            if pose_frame_indices[pose_idx] == frame_idx:
                current_pose_idx = pose_idx
                pose_idx += 1
                break
            elif pose_frame_indices[pose_idx] < frame_idx:
                pose_idx += 1
            else:
                break

        context = LayerContext(
            frame_width=draw_w,
            frame_height=draw_h,
            fps=meta.fps,
            frame_idx=frame_idx,
            total_frames=total,
            normalized=True,
        )

        if layer >= 0 and current_pose_idx is not None:
            foot_kp = raw_foot_kps[current_pose_idx] if raw_foot_kps is not None else None
            frame = draw_skeleton(
                frame, poses[current_pose_idx], draw_h, draw_w,
                line_width=1, joint_radius=3, foot_keypoints=foot_kp,
            )
            context.pose_2d = poses_viz[current_pose_idx]
            if poses_3d is not None and current_pose_idx < len(poses_3d):
                context.pose_3d = poses_3d[current_pose_idx]

        if layer >= 1 and current_pose_idx is not None:
            frame = render_layers(frame, layers, context)

        # HUD: frame counter + timestamp
        minutes = int(frame_idx / meta.fps) // 60
        seconds = int(frame_idx / meta.fps) % 60
        ms = int((frame_idx / meta.fps - int(frame_idx / meta.fps)) * 100)
        frame_text = f"{frame_idx}/{total}  {minutes:02d}:{seconds:02d}.{ms:02d}"
        put_text(frame, frame_text, (draw_w - 220, 30), font_size=18, color=(200, 200, 200))

        # Export data
        if export and current_pose_idx is not None:
            export_frames.append(frame_idx)
            export_timestamps.append(round(frame_idx / meta.fps, 3))
            export_floor_angles.append(0.0)
            ja = compute_joint_angles(poses_viz[current_pose_idx])
            export_joint_angles.append(ja)
            export_poses_list.append(poses[current_pose_idx].copy())

        writer.write(frame)
        frame_idx += 1

        if progress_cb and frame_idx % 50 == 0:
            progress_cb(0.3 + 0.65 * frame_idx / total, f"Rendering frame {frame_idx}/{total}")

    cap.release()
    writer.release()

    if progress_cb:
        progress_cb(0.95, "Saving exports...")

    # Export NPY + CSV
    poses_path = None
    csv_path = None
    if export and export_poses_list:
        out_dir = output_path.parent
        stem = output_path.stem

        poses_path = out_dir / f"{stem}_poses.npy"
        np.save(str(poses_path), np.array(export_poses_list))

        csv_path = out_dir / f"{stem}_biomechanics.csv"
        angle_keys = [
            "R Ankle", "L Ankle", "R Knee", "L Knee",
            "R Hip", "L Hip", "R Shoulder", "L Shoulder",
            "R Elbow", "L Elbow", "R Wrist", "L Wrist",
        ]
        header = ["frame", "timestamp_s", "floor_angle_deg"] + angle_keys
        with open(csv_path, "w", newline="") as f:
            w = _csv.writer(f)
            w.writerow(header)
            for idx in range(len(export_frames)):
                ja = export_joint_angles[idx]
                row = [export_frames[idx], export_timestamps[idx], export_floor_angles[idx]]
                row += [round(ja.get(k, float("nan")), 1) for k in angle_keys]
                w.writerow(row)

    return {
        "video_path": str(output_path),
        "poses_path": str(poses_path) if poses_path else None,
        "csv_path": str(csv_path) if csv_path else None,
        "stats": {
            "total_frames": total,
            "valid_frames": n_valid,
            "fps": meta.fps,
            "resolution": f"{meta.width}x{meta.height}",
        },
    }
```

```python
#!/usr/bin/env python3
"""Gradio UI for figure skating biomechanics analysis.

Wraps the visualization pipeline with interactive person selection.

Usage:
    uv run python scripts/gradio_app.py
    # or after install:
    skating-gradio
"""

from __future__ import annotations

import tempfile
from pathlib import Path

import cv2
import gradio as gr
import numpy as np

from src.gradio_helpers import (
    choice_to_person_click,
    match_click_to_person,
    persons_to_choices,
    process_video_pipeline,
    render_person_preview,
)
from src.types import PersonClick
from src.utils.video import get_video_meta


def _detect_persons(video_path: str, tracking: str):
    """Step 1: Extract first 30 frames and detect all persons."""
    if not video_path:
        return None, gr.update(choices=[], value=None), [], "Upload a video first."

    from src.pose_estimation.rtmlib_extractor import RTMPoseExtractor

    extractor = RTMPoseExtractor(
        output_format="normalized",
        conf_threshold=0.3,
        det_frequency=1,
        device="cuda",
        tracking_mode=tracking,
    )

    persons, preview_path = extractor.preview_persons(video_path)

    if not persons:
        return None, gr.update(choices=[], value=None), [], "No persons detected."

    # Read first frame for annotation
    cap = cv2.VideoCapture(video_path)
    ret, frame = cap.read()
    cap.release()
    if not ret:
        return None, gr.update(choices=[], value=None), [], "Cannot read video."

    # Render numbered bounding boxes
    annotated = render_person_preview(frame, persons)

    # Build radio choices
    choices = persons_to_choices(persons)

    # Auto-select if only 1 person
    auto_value = choices[0] if len(choices) == 1 else None

    status = f"Detected {len(persons)} person(s). Click on the image or select below."

    return (
        annotated,
        gr.update(choices=choices, value=auto_value),
        persons,
        status,
    )


def _on_image_select(evt: gr.SelectData, persons_state):
    """Handle click on preview image — match to person bbox."""
    if not persons_state:
        return "No persons detected yet.", None

    # evt.index gives pixel coordinates
    # We need to convert to normalized
    # But we don't know the original image dimensions from here easily
    # Use the annotated image dimensions
    return f"Clicked at ({evt.index[0]}, {evt.index[1]}). Select from the dropdown below.", None


def _on_person_select(choice: str, persons_state, video_path):
    """Handle person selection from Radio — highlight on preview."""
    if not choice or not persons_state or not video_path:
        return None, None

    idx = int(choice.split("#")[1].split(" ")[0]) - 1

    cap = cv2.VideoCapture(video_path)
    ret, frame = cap.read()
    cap.release()
    if not ret:
        return None, None

    annotated = render_person_preview(frame, persons_state, selected_idx=idx)

    meta = get_video_meta(video_path)
    person_click = choice_to_person_click(choice, persons_state, meta.width, meta.height)

    return annotated, person_click


def _run_pipeline(
    video_path: str,
    person_click_state,
    persons_state,
    person_choice: str,
    layer: int,
    tracking: str,
    use_3d: bool,
    render_scale: float,
    export: bool,
    progress=gr.Progress(),
):
    """Step 3: Run the full visualization pipeline."""
    if not video_path:
        return None, None, None, "No video provided."

    # Resolve person_click
    person_click: PersonClick | None = person_click_state
    if person_click is None and persons_state and person_choice:
        meta = get_video_meta(video_path)
        person_click = choice_to_person_click(
            person_choice, persons_state, meta.width, meta.height,
        )

    # Output path in temp dir
    output_dir = Path(tempfile.mkdtemp(prefix="skating_"))
    video_name = Path(video_path).stem
    output_path = output_dir / f"{video_name}_layer{layer}.mp4"

        progress(frac, desc=desc)

        video_path=video_path,
        person_click=person_click,
        layer=layer,
        tracking=tracking,
        use_3d=use_3d,
        render_scale=render_scale,
        blade_3d=False,
        export=export,
        output_path=str(output_path),
        progress_cb=progress_cb,
    )

    status = (

    )

    poses_file = result["poses_path"]
    csv_file = result["csv_path"]

    return (
        result["video_path"],
        poses_file,
        csv_file,
        status,
    )
```

```python
def _on_image_select(evt: gr.SelectData, persons_state, video_path):
    """Handle click on preview image — match to person bbox."""
    if not persons_state or not video_path:
        return "No persons detected yet.", None, None

    meta = get_video_meta(video_path)
    # evt.index is pixel coordinates on the displayed image
    # Convert to normalized [0,1]
    x_norm = evt.index[0] / meta.width
    y_norm = evt.index[1] / meta.height

    matched = match_click_to_person(persons_state, x_norm, y_norm)
    if matched is None:
        return (
            f"Click ({evt.index[0]}, {evt.index[1]}) — no person at that location. Try clicking on a person.",
            None,
            None,
        )

    # Find index of matched person
    idx = next(i for i, p in enumerate(persons_state) if p["track_id"] == matched["track_id"])

    cap = cv2.VideoCapture(video_path)
    ret, frame = cap.read()
    cap.release()
    if not ret:
        return "Cannot read video.", None, None

    annotated = render_person_preview(frame, persons_state, selected_idx=idx)
    person_click = PersonClick(
        x=int(matched["mid_hip"][0] * meta.width),
        y=int(matched["mid_hip"][1] * meta.height),
    )

    return (
        f"Selected Person #{idx + 1} (track {matched['track_id']}, {matched['hits']} hits)",
        annotated,
        person_click,
    )
```
